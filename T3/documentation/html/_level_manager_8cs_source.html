148 // Calculate score based on the number of completed tasks

150 // if the player didn't complete all the tasks, reduce the score
151 if (finishedTasks < totalTasks)

153 score -= (totalTasks - finishedTasks) * 50;

164 score += repairedServers * 300;
165
166 // Calculate score based on the number of backup data
167 score += backupData * 200;
168
169 // Calculate score based on the time remaining
170 // for each second remaining, add 10 points
171 // but there is a problem the player can just finish the level in the game start (480 * 10 = 4800)
172 // so we need to not give points if the player finishes the level in less than 5 minutes
173 if (_chrono < 200)
174 {
175 score += (int)(_chrono * 10);
176 }
177
178 // Get the player's precision
179 float precision = _player.GetAttributeValue(Attribute.Precision);
180
181 // the more precise the player is, the more points they get
182 score += (int)(precision * 50);
183
